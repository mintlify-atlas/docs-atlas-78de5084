## getHistoriaPageQuery

Retrieves all content for the Historia (history) page including background images, gallery items, and multiple content sections.

### Function Signature

```typescript
getHistoriaPageQuery(lang: string = 'es'): string
```

### Parameters

<ParamField path="lang" type="string" default="es">
  Language code for localized content. Supports 'es', 'ca', 'en', etc. Falls back to Spanish if translation not available.
</ParamField>

### GROQ Query

```groq
*[_type == "historiaPage" && _id == "historiaPage"][0] {
  backgroundImages {
    desktop,
    mobile
  },
  "headHighlightedText": {
    "desktop": coalesce(headHighlightedText.desktop.ca, headHighlightedText.desktop.es, ""),
    "tablet": coalesce(headHighlightedText.tablet.ca, headHighlightedText.tablet.es, ""),
    "mobile": coalesce(headHighlightedText.mobile.ca, headHighlightedText.mobile.es, "")
  },
  "sectionTwoGallery": sectionTwoGallery[] {
    "url": image.asset->url,
    "alt": image.alt,
    "title": coalesce(title.ca, title.es, ""),
    "text": coalesce(text.ca, text.es, "")
  },
  "sectionTwoHighlightedText": {
    "desktop": coalesce(sectionTwoHighlightedText.desktop.ca, sectionTwoHighlightedText.desktop.es, ""),
    "tablet": coalesce(sectionTwoHighlightedText.tablet.ca, sectionTwoHighlightedText.tablet.es, ""),
    "mobile": coalesce(sectionTwoHighlightedText.mobile.ca, sectionTwoHighlightedText.mobile.es, "")
  },
  sectionThreeImages {
    image1,
    image2,
    image3
  },
  sectionThreeMobileImage,
  floatingImage,
  sectionFourHighlightedText {
    "desktop": coalesce(desktop.ca, desktop.es, ""),
    "mobile": coalesce(mobile.ca, mobile.es, "")
  },
  sectionFourImage,
  "sectionFourBodyText": coalesce(sectionFourBodyText.ca, sectionFourBodyText.es, ""),
  sectionFourImages {
    image1,
    image2,
    image3,
    image4
  },
  sectionFiveColumn1 {
    highlightedText {
      "desktop": coalesce(desktop.ca, desktop.es, ""),
      "tablet": coalesce(tablet.ca, tablet.es, ""),
      "mobile": coalesce(mobile.ca, mobile.es, "")
    },
    "bodyText": coalesce(bodyText.ca, bodyText.es, ""),
    "buttonText": coalesce(buttonText.ca, buttonText.es, ""),
    buttonUrl
  },
  sectionFiveColumn2 {
    backgroundImage,
    image1,
    image2,
    image3
  },
  sectionLastHistoriaColumn1 {
    highlightedText {
      "desktop": coalesce(desktop.ca, desktop.es, ""),
      "tablet": coalesce(tablet.ca, tablet.es, ""),
      "mobile": coalesce(mobile.ca, mobile.es, "")
    },
    "bodyText": coalesce(bodyText.ca, bodyText.es, ""),
    "buttonText": coalesce(buttonText.ca, buttonText.es, ""),
    buttonUrl
  },
  sectionLastHistoriaColumn2 {
    backgroundImage
  },
  "scrollHorizontalHighlightedText": {
    "desktop": coalesce(scrollHorizontalHighlightedText.desktop.ca, scrollHorizontalHighlightedText.desktop.es, ""),
    "tablet": coalesce(scrollHorizontalHighlightedText.tablet.ca, scrollHorizontalHighlightedText.tablet.es, ""),
    "mobile": coalesce(scrollHorizontalHighlightedText.mobile.ca, scrollHorizontalHighlightedText.mobile.es, "")
  },
  "scrollHorizontalImage": scrollHorizontalImage.asset->url,
  "scrollHorizontalImageAlt": scrollHorizontalImage.alt,
  "scrollHorizontalItems": scrollHorizontalItems[] {
    "imageUrl": image.asset->url,
    "imageAlt": image.alt,
    "text": coalesce(text.ca, text.es, "")
  }
}
```

<ResponseField name="backgroundImages" type="object">

<ResponseField name="sectionFiveColumn2" type="object">

<ResponseField name="sectionLastHistoriaColumn2" type="object">
  Final section right column with background image

### Usage Example

```typescript
import { sanityClient } from '@/lib/sanity';
import { getHistoriaPageQuery } from '@/lib/sanityQueries';

// In an Astro page component
const lang = Astro.currentLocale || 'es';
const historiaData = await sanityClient.fetch(getHistoriaPageQuery(lang));
```

```astro
---
// pages/historia.astro
import { sanityClient } from '@/lib/sanity';
import { getHistoriaPageQuery } from '@/lib/sanityQueries';

const lang = Astro.currentLocale || 'es';
const data = await sanityClient.fetch(getHistoriaPageQuery(lang));
---

<section>
  <h1>{data.headHighlightedText.desktop}</h1>
  
  <div class="gallery">
    {data.sectionTwoGallery.map(item => (
      <div>
        <img src={item.url} alt={item.alt} />
        <h3>{item.title}</h3>
        <p>{item.text}</p>
      </div>
    ))}
  </div>
  
  <div class="horizontal-scroll">
    {data.scrollHorizontalItems.map(item => (
      <div>
        <img src={item.imageUrl} alt={item.imageAlt} />
        <p>{item.text}</p>
      </div>
    ))}
  </div>
</section>
```
