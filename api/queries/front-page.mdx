## getFrontPageQuery

Retrieves all content for the homepage including hero images, marquee text, sections, and club signup content.

### Function Signature

```typescript
getFrontPageQuery(lang: string = 'es'): string
```

### Parameters

<ParamField path="lang" type="string" default="es">
  Language code for localized content. Supports 'es', 'ca', 'en', etc. Falls back to Spanish if translation not available.
</ParamField>

### GROQ Query

```groq
*[_type == "frontPage" && _id == "frontPage"][0] {
  backgroundImages {
    desktop,
    mobile
  },
  orderLink,
  "marqueeText": coalesce(marqueeText.ca, marqueeText.es, ""),
  images {
    image1,
    image2,
    image3
  },
  "floatingTextLeft": coalesce(floatingTextLeft.ca, floatingTextLeft.es, ""),
  "floatingTextRight": coalesce(floatingTextRight.ca, floatingTextRight.es, ""),
  highlightedText {
    "desktop": coalesce(desktop.ca, desktop.es, []),
    "tablet": coalesce(tablet.ca, tablet.es, []),
    "mobile": coalesce(mobile.ca, mobile.es, [])
  },
  "bodyText": coalesce(bodyText.ca, bodyText.es, ""),
  sectionThreeImages {
    image1,
    image2,
    image3
  },
  floatingImage,
  sectionFourHighlightedText {
    "desktop": coalesce(desktop.ca, desktop.es, ""),
    "mobile": coalesce(mobile.ca, mobile.es, "")
  },
  sectionFourImage,
  "sectionFourBodyText": coalesce(sectionFourBodyText.ca, sectionFourBodyText.es, ""),
  sectionFourImages {
    image1,
    image2,
    image3,
    image4
  },
  sectionFiveColumn1 {
    highlightedText {
      "desktop": coalesce(desktop.ca, desktop.es, ""),
      "tablet": coalesce(tablet.ca, tablet.es, ""),
      "mobile": coalesce(mobile.ca, mobile.es, "")
    },
    "bodyText": coalesce(bodyText.ca, bodyText.es, ""),
    "buttonText": coalesce(buttonText.ca, buttonText.es, ""),
    buttonUrl
  },
  sectionFiveColumn2 {
    backgroundImage,
    image1,
    image2,
    image3
  },
  sectionClubHighlightedText {
    "desktop": coalesce(desktop.ca, desktop.es, ""),
    "tablet": coalesce(tablet.ca, tablet.es, ""),
    "mobile": coalesce(mobile.ca, mobile.es, "")
  },
  "sectionClubBodyText": coalesce(sectionClubBodyText.ca, sectionClubBodyText.es, ""),
  "sectionClubEmailPlaceholder": coalesce(sectionClubEmailPlaceholder.ca, sectionClubEmailPlaceholder.es, ""),
  "sectionClubButtonText": coalesce(sectionClubButtonText.ca, sectionClubButtonText.es, "")
}
```

### Usage Example

```typescript
import { sanityClient } from '@/lib/sanity';
import { getFrontPageQuery } from '@/lib/sanityQueries';

// In an Astro page component
const lang = Astro.currentLocale || 'es';
const frontPageData = await sanityClient.fetch(getFrontPageQuery(lang));
```

```astro
---
// pages/index.astro
import { sanityClient } from '@/lib/sanity';
import { getFrontPageQuery } from '@/lib/sanityQueries';

const lang = Astro.currentLocale || 'es';
const data = await sanityClient.fetch(getFrontPageQuery(lang));
---

<section>

</section>
```
