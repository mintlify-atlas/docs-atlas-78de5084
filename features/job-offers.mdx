**File:** `src/pages/talento.astro:543-558`

```javascript
function filterJobs(searchTerm: string) {
  const term = searchTerm.toLowerCase().trim();
  
  if (!term) {
    filteredJobs = [...allJobs];
  } else {
    filteredJobs = allJobs.filter(job =>
      job.title.toLowerCase().includes(term) ||
      job.description.toLowerCase().includes(term) ||
      job.province.name.toLowerCase().includes(term) ||
      (job.locality && job.locality.toLowerCase().includes(term))
    );
  }
  
  renderJobs();
}
```

**Search targets:**
- Job title
- Job description
- Province name
- Locality/city name

```javascript
function groupJobsByProvince(jobs: any[]) {
  const grouped: Record<string, any[]> = {};
  
  jobs.forEach(job => {
    const provinceName = job.province.name;
    if (!grouped[provinceName]) {
      grouped[provinceName] = [];
    }
    grouped[provinceName].push(job);
  });
  
  return grouped;
}
```

### Accordion Functionality

**File:** `src/pages/talento.astro:493-541`

```javascript
function initAccordions() {
  const cards = provincesList.querySelectorAll('.job-card');
  
  cards.forEach(card => {
    const header = card.querySelector('.job-card-header');
    const body = card.querySelector('.job-card-body');
    const icon = card.querySelector('.job-title .icon');
    
    // First card open by default
    if (card.classList.contains('is-open')) {
      gsap.set(body, { height: 'auto', overflow: 'hidden' });
      if (icon) gsap.set(icon, { rotation: 90 });
    }
    
    header.addEventListener('click', () => {
      const isOpen = card.classList.contains('is-open');
      
      if (isOpen) {
        // Close
        gsap.to(body, { height: 0, duration: 0.4, ease: 'power2.inOut' });
        gsap.to(icon, { rotation: 0, duration: 0.3, ease: 'power2.inOut' });
        card.classList.remove('is-open');
      } else {
        // Open
        const targetHeight = body.offsetHeight;
        gsap.fromTo(body, { height: 0 }, { height: targetHeight, duration: 0.4, ease: 'power2.inOut' });
        gsap.to(icon, { rotation: 90, duration: 0.3, ease: 'power2.inOut' });
        card.classList.add('is-open');
      }
    });
  });
}
```

**Animation:**
- Body height: 0 ↔ auto (0.4s, power2.inOut)
- Icon rotation: 0° ↔ 90° (0.3s, power2.inOut)

```javascript
const fileInputs = document.querySelectorAll('.cv__field--file input[type="file"]');
fileInputs.forEach(input => {
  input.addEventListener('change', (e) => {
    const target = e.target as HTMLInputElement;
    const box = target.closest('.cv__field--file')?.querySelector('.cv__file-box');
    const nameSpan = box?.querySelector('.cv__file-name');
    const maxSpan = box?.querySelector('.cv__file-max');
    
    if (nameSpan) {
      const fileName = target.files?.[0]?.name || '';
      nameSpan.textContent = fileName;
      nameSpan.classList.toggle('has-file', !!fileName);
      
      if (maxSpan) {
        maxSpan.style.display = fileName ? 'none' : '';
      }
    }
  });
});
```

### Form Submission

**File:** `src/pages/talento.astro:614-661`

```javascript
cvForm.addEventListener('submit', async (e) => {
  e.preventDefault();
  
  const btn = cvForm.querySelector('button[type="submit"]');
  const originalText = btn.textContent;
  btn.disabled = true;
  btn.textContent = txtSending;
  
  try {
    const formData = new FormData(cvForm);
    const { data, error } = await actions.submitCvForm(formData);
    
    let msgDiv = cvForm.querySelector('.cv__message');
    if (!msgDiv) {
      msgDiv = document.createElement('div');
      msgDiv.className = 'cv__message';
      cvForm.appendChild(msgDiv);
    }
    
    if (error || !data?.success) {
      msgDiv.className = 'cv__message error';
      msgDiv.textContent = data?.message || txtError;
    } else {
      msgDiv.className = 'cv__message success';
      msgDiv.textContent = data.message || txtSuccess;
      cvForm.reset();
      // Reset file inputs UI
    }
  } catch (err) {
    console.error(err);
  } finally {
    btn.disabled = false;
    btn.textContent = originalText;
  }
});
```
